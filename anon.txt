67. INT binaryToDecimal(char* binary) {
int sum = 0, length = strlen(binary), powerOfTwo = 1;
for (int i = length - 1; i >= 0; i--) {
sum += (binary[i] -'0') * powerOfTwo;
powerOfTwo *= 2;
}
return sum;
}
